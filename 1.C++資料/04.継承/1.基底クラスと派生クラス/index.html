class Car
{
public:
    Car() {}
    ~Car() {}

public:
    // гӮЁгғігӮёгғігӮ’е§ӢеӢ•гҒҷгӮӢ
    void startEngine()
    {
        std::cout << "гӮЁгғігӮёгғігӮ’е§ӢеӢ•гҒҷгӮӢ" << std::endl;
        running_ = true;
    }

    // гӮЁгғігӮёгғігӮ’еҒңжӯўгҒҷгӮӢ
    void stopEngine()
    {
        std::cout << "гӮЁгғігӮёгғігӮ’еҒңжӯўгҒҷгӮӢ" << std::endl;
        running_ = false;
    }


public:
    bool running_ = false; // гӮЁгғігӮёгғігҒҢеӢ•гҒ„гҒҰгҒ„гӮӢ
};

// гӮҝгғігӮҜгғӯгғјгғӘгғјгӮҜгғ©гӮ№
class Tank : public Car
{
public:
    Tank() {}
    ~Tank() {}

public:
    // зҮғж–ҷгӮ’йҖҒеҮәгҒҷгӮӢ
    void pumpOut()
    {
        // running_ гӮ’е®ЈиЁҖгҒ—гҒҰгҒ„гӮӢгҒ®гҒҜ Car гӮҜгғ©гӮ№гҒ гҒҢгҖҒ
        // Tank гӮҜгғ©гӮ№гҒҢ Car гӮҜгғ©гӮ№гӮ’з¶ҷжүҝгҒ—гҒҰгҒ„гӮӢзӮәеҲ©з”ЁгҒ§гҒҚгӮӢ
        if (running_) {
            std::cout << "гӮЁгғігӮёгғігҒҢеӢ•гҒ„гҒҰгҒ„гӮӢгҒ®гҒ§зҮғж–ҷгҒ®йҖҒеҮәгҒҜеҚұйҷә" << std::endl;
        }
        else
        {
            std::cout << "зҮғж–ҷгӮ’йҖҒеҮәгҒҷгӮӢ" << std::endl;
        }
    }

};

// гӮЁгғігғҲгғӘгғјй–ўж•°
int main()
{
    Car car;
    // и»ҠгҒ®гӮЁгғігӮёгғігӮ’е§ӢеӢ•гҒҷгӮӢ
    car.startEngine();

    Tank tank;
    // startEngine() гҒЁ stopEngine() гӮ’е®ҡзҫ©гҒ—гҒҰгҒ„гӮӢгҒ®гҒҜ Car гӮҜгғ©гӮ№гҒ гҒҢгҖҒ
    // Tank гӮҜгғ©гӮ№гҒҢ Car гӮҜгғ©гӮ№гӮ’з¶ҷжүҝгҒ—гҒҰгҒ„гӮӢзӮәеҲ©з”ЁгҒ§гҒҚгӮӢ
    tank.startEngine();
    tank.stopEngine();

    // еҪ“з„¶ Tank гӮҜгғ©гӮ№гҒ§е®ҡзҫ©гҒ—гҒҹй–ўж•°гҒҜе‘јгҒіеҮәгҒӣгӮӢ
    tank.pumpOut();
}
